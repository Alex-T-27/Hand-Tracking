# HandTracking — pinned, conflict-free dependency set (Python 3.12).
#
# Recreate the environment from scratch with:
#   python -m venv venv
#   source venv/bin/activate
#   pip install -r requirements.txt
#
# WHY THESE PINS:
# - mediapipe 0.10.21 requires numpy<2. If numpy is left unpinned, pip grabs
#   the latest numpy (2.x) and mediapipe breaks ("module 'mediapipe' has no
#   attribute 'solutions'"). numpy==1.26.4 is the last 1.x and satisfies
#   mediapipe (<2) AND opencv-contrib (>=1.26.0) AND jax (>=1.26).
# - jax/jaxlib/scipy/ml_dtypes are mediapipe's transitive deps that depend on
#   numpy. They're pinned so a FUTURE release can't raise the numpy floor to
#   >=2 and silently re-break the build.
# - opencv-contrib-python ONLY. Never add opencv-python — they share the cv2/
#   directory and uninstalling one corrupts the other.

# --- direct deps ---
mediapipe==0.10.21
opencv-contrib-python==4.11.0.86

# --- numpy-sensitive lock (the chain that caused the conflicts) ---
numpy==1.26.4
jax==0.7.1
jaxlib==0.7.1
scipy==1.17.1
ml_dtypes==0.5.4
